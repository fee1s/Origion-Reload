SOV_industry_relocation = {
	SOV_auto_industry_relocation_first_wave_mission = {
		icon = generic_construction
		
		days_mission_timeout = 7
		
		is_good = yes 
		
		fire_only_once = yes
		allowed = {
			always = no
		}
		available = {
			custom_trigger_tooltip = {
				tooltip = SOV_evacuation_availible_tt
				NOT = {
					any_controlled_state = {
						SOV_is_industrial_relocation_first_wave = yes
						NOT = { has_state_flag = SOV_after_combat } 
						NOT = { has_state_flag = SOV_evacuation_finished_flag }
					}
				}
			}
		}

			timeout_effect = {
				custom_effect_tooltip = SOV_evacuation_header_tt
				if = {
					limit = {
						any_owned_state = {
							SOV_is_industrial_relocation_first_wave = yes
							NOT = { has_state_flag = SOV_evacuation_finished_flag }
							NOT = { has_state_flag = SOV_after_combat }
							NOT = { is_fully_controlled_by = SOV }
						}
						
					}
					custom_effect_tooltip = generic_skip_one_line_tt
					custom_effect_tooltip = SOV_evacuation_red_first_wave_tt
				}
				if = {
					limit = {
						any_owned_state = {
							SOV_is_industrial_relocation_first_wave = yes
							NOT = { has_state_flag = SOV_evacuation_finished_flag }
							NOT = { has_state_flag = SOV_after_combat }
							is_fully_controlled_by = SOV
							any_neighbor_state = {
								controller = { has_war_with = SOV }
							}
						}
					}
					custom_effect_tooltip = generic_skip_one_line_tt
					custom_effect_tooltip = SOV_evacuation_yellow_first_wave_tt
				}
				if = {
					limit = {
						any_owned_state = {
							SOV_is_industrial_relocation_first_wave = yes
							NOT = { has_state_flag = SOV_evacuation_finished_flag }
							NOT = { has_state_flag = SOV_after_combat }
							is_fully_controlled_by = SOV
							NOT = {	any_neighbor_state = { controller = { has_war_with = SOV } } }
						}
					}
					custom_effect_tooltip = generic_skip_one_line_tt
					custom_effect_tooltip = SOV_evacuation_green_first_wave_tt
				}
			hidden_effect = {
			
				set_temp_variable = { SOV_removed_mils_count = 0 }
				set_temp_variable = { SOV_removed_civs_count = 0 }
			
				every_controlled_state = {
					limit = {
						SOV_is_industrial_relocation_first_wave = yes
						NOT = { has_state_flag = SOV_after_combat } 
						NOT = { has_state_flag = SOV_evacuation_finished_flag }
					}
					
					if = {
						limit = { arms_factory > 0 }
						
						remove_building = { type = arms_factory level = 1 }
						
						ROOT = { add_to_temp_variable = { SOV_removed_mils_count = 1 } }
					}
					else_if = {
						limit = { industrial_complex > 0 }
						remove_building = { type = industrial_complex level = 1 }
						ROOT = { add_to_temp_variable = { SOV_removed_civs_count = 1 } }
					}
					
					if = {
						limit = {
							arms_factory < 1
							industrial_complex < 1
						}
						set_state_flag = SOV_evacuation_finished_flag
					}
				}

				ROOT = {
					while_loop_effect = {
						limit = { check_variable = { SOV_removed_mils_count > 0 } }
						
						if = {
							limit = {
								any_controlled_state = { 
									SOV_Ural_ir_place = yes 
									free_building_slots = { building = arms_factory size > 0 include_locked = yes }
								}
							}
							random_controlled_state = {
								limit = {
									SOV_Ural_ir_place = yes 
									free_building_slots = { building = arms_factory size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = arms_factory level = 1 instant_build = yes }
							}
						}
						else_if = {
							limit = {
								any_controlled_state = { 
									SOV_Kazah_ir_place = yes 
									free_building_slots = { building = arms_factory size > 0 include_locked = yes }
								}
							}
							random_controlled_state = {
								limit = {
									SOV_Kazah_ir_place = yes 
									free_building_slots = { building = arms_factory size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = arms_factory level = 1 instant_build = yes }
							}
						}
						else = {
							random_controlled_state = {
								limit = { 
									is_on_continent = asia 
									free_building_slots = { building = arms_factory size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = arms_factory level = 1 instant_build = yes }
							}
						}
						
						subtract_from_temp_variable = { SOV_removed_mils_count = 1 }
					}
					while_loop_effect = {
						limit = { check_variable = { SOV_removed_civs_count > 0 } }
						
						if = {
							limit = {
								any_controlled_state = { 
									SOV_Ural_ir_place = yes 
									free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
								}
							}
							random_controlled_state = {
								limit = {
									SOV_Ural_ir_place = yes 
									free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
							}
						}
						else_if = {
							limit = {
								any_controlled_state = { 
									SOV_Kazah_ir_place = yes 
									free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
								}
							}
							random_controlled_state = {
								limit = {
									SOV_Kazah_ir_place = yes 
									free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
							}
						}
						else = {
							random_controlled_state = {
								limit = { 
									is_on_continent = asia 
									free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
								}
								add_extra_state_shared_building_slots = 1
								add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
							}
						}
						
						subtract_from_temp_variable = { SOV_removed_civs_count = 1 }
					}
					
					if = {
						limit = {
							any_controlled_state = {
								SOV_is_industrial_relocation_first_wave = yes
								NOT = { has_state_flag = SOV_after_combat } 
								NOT = { has_state_flag = SOV_evacuation_finished_flag }
							}
						}
						activate_mission = SOV_auto_industry_relocation_first_wave_mission
					}
				}
			}
		}
	}
